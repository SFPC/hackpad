Defne’s Shader:

code:

#version 120

void main( void )

{

    float width = 1024.0;

    float height = 768.0;

        float x = gl_FragCoord.x / width;

    float y = gl_FragCoord.y / height;

    float red, green, blue;

    

    if (x > 0.5) {

        red = sin(x * (40 * x *x));

        green = sin(x * (20 * x *x));

        blue = sin(x * (10 * x *x));

    }

    

    

    gl_FragColor = vec4(red, green, blue, 1.0);

}

another:

#version 120

void main(void)

{

        float width = 1024.0;

        float height = 768.0;

        float x = gl_FragCoord.x / width;

        float y = gl_FragCoord.y / height;

        float red, green, blue;

        if (tan(x * 30) > cos(y * 50)) {

                red = sin(x / (y + 1));

                blue = cos(y - x);

                green = tan(x);

        } 

        else {

                red = tan(y + x);

                blue = cos(sin(x * y));

                green = sin(tan(x + y));

        }

        gl_FragColor = vec4(red, green, blue, 1.0);

        //gl_FragColor = vec4(1.0, 0.0, 0.0, 1.0);

}
